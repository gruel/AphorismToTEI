[docs]    def save_xml(self, fname=None, xml=None):
        """Method to save the XML in the working directory
        """
        if fname is None:
            fname = self.__class__.__name__ + '.xml'

        if xml is None:
            if self.xml is list:
                xml = '\n'.join(self.xml)
            else:
                xml = self.xml

        with open(fname, 'w', encoding="utf-8") as f:
            f.write(xml)
